interpret <- function(num_views){
  if (num_views > 15) {
    print('You\'re popular!')
    return(num_views)
  } else {
    print('Try to be more visible!')
    return(0)

# Define the interpret_all() function
interpret_all <- function(data, logi = TRUE){
  yy <- 0
  for (i in data) {
    yy <- yy + interpret(i)
  }
  if (logi) {
    return(yy)
  } else {
    return(NULL)
  }
}
